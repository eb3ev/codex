#== Assignment

# Operator | Example | Same as
# -------- | ------- | ----------
=        # | x = 1   | x = 1
+=       # | x += 1  | x = x + 1
-=       # | x -= 1  | x = x - 1
*=       # | x *= 1  | x = x * 1
/=       # | x /= 1  | x = x / 1
%=       # | x %= 1  | x = x % 1
//=      # | x //= 1 | x = x // 1
**=      # | x **= 1 | x = x ** 1
&=       # | x &= 1  | x = x & 1
|=       # | x |= 1  | x = x | 1
^=       # | x ^= 1  | x = x ^ 1
>>=      # | x >>= 1 | x = x >> 1
<<=      # | x <<= 1 | x = x << 1
